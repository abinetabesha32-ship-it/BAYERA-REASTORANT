// Utility: Render stars
function renderStars(rating, fixed = true) {
    const max = 5;
    let html = '';
    for (let i = 1; i <= max; i++) {
        if (i <= Math.floor(rating)) html += `<i class="fas fa-star"></i>`;
        else if (!fixed && i === Math.ceil(rating) && rating % 1 !== 0) html += `<i class="fas fa-star-half-alt"></i>`;
        else html += `<i class="far fa-star"></i>`;
    }
    return html;
}

    const cards = menuData[category];

}

// --- CART LOGIC ---
let cart = [];
function addToCart(category, idx) {
    const item = menuData[category][idx];
    // Check if already exists
    const found = cart.find(c => c.name === item.name && c.category === category);
    if(found) found.qty += 1;
    else cart.push({...item, category, qty: 1});
    updateCart();
}
document.getElementById('food-cards').addEventListener('click', function(e){
    if(e.target.classList.contains('add-btn')){
        const {category, index} = e.target.dataset;
        addToCart(category, parseInt(index));
        // Animate button
        e.target.classList.add('btn-added');
        setTimeout(() => e.target.classList.remove('btn-added'), 300);
    }
});
function updateCart() {
    const cartList = document.getElementById('cart-items');
    cartList.innerHTML = '';
    let total = 0;
    cart.forEach((item, idx) => {
        const li = document.createElement('li');
        li.className = 'cart-item';
        li.innerHTML = `
            ${item.name} x${item.qty} 
            <span> $${(item.price * item.qty).toFixed(2)} 
            <span class="remove" data-index="${idx}"><i class="fas fa-times-circle"></i></span></span>
        `;
        cartList.appendChild(li);
        total += item.price * item.qty;
    });
    document.getElementById('cart-total').textContent = total.toFixed(2);

    // Remove items
    cartList.querySelectorAll('.remove').forEach(rm => {
        rm.addEventListener('click', function(){
            const i = parseInt(rm.dataset.index);
            cart.splice(i,1);
            updateCart();
        });
    });
}
document.getElementById('place-order-btn').addEventListener('click', function() {
    if(cart.length === 0){
        alert("No items in cart!");
        return;
    }
    alert("Order placed! Thank you for choosing BAYYERA RESTAURANT.");
    cart = [];
    updateCart();
});
